**Create server/middleware/requireAuth.cjs:**

```javascript
'use strict';

// requireAuth — rejects requests that have no hs_refresh session cookie.
// This is a presence check (not a signature verification) — it confirms the
// user went through the login flow. Full JWT signature verification is deferred
// until the WC JWT secret is available as a server env var.
function requireAuth(req, res, next) {
  const token = req.cookies?.hs_refresh;
  if (!token) {
    return res.status(401).json({ error: true, code: 'UNAUTHENTICATED', message: 'Authentication required' });
  }
  next();
}

module.exports = requireAuth;
```
